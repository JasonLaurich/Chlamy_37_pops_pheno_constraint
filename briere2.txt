model {
  # Priors for parameters
  cf.a ~ dunif(0, 10)          # Uniform prior for scaling factor
  cf.tmin ~ dunif(0, 10)       # Uniform prior for lower temperature bound
  cf.tmax ~ dunif(25, 50)      # Uniform prior for upper temperature bound
  cf.b ~ dunif(1, 5)           # Uniform prior for exponent parameter
  cf.sigma ~ dunif(0, 5)       # Prior for standard deviation
  cf.tau <- pow(cf.sigma, -2)  # Precision (inverse variance)

  # Likelihood
  for (i in 1:N.obs) {
    trait[i] ~ dnorm(trait.mu[i], cf.tau)  # Observed trait
    trait.mu[i] <- cf.a * temp[i] * (temp[i] - cf.tmin) *
                   pow((cf.tmax - temp[i]), 1 / cf.b) *
                   step(temp[i] - cf.tmin) * step(cf.tmax - temp[i])  # Model equation with bounds
  }

  # Predictions for new temperatures
  for (j in 1:N.Temp.xs) {
    r.pred[j] <- cf.a * Temp.xs[j] * (Temp.xs[j] - cf.tmin) *
                 pow((cf.tmax - Temp.xs[j]), 1 / cf.b) *
                 step(Temp.xs[j] - cf.tmin) * step(cf.tmax - Temp.xs[j])
  }
}
