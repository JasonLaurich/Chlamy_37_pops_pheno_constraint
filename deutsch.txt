model {

  ## Priors for parameters
  rmax ~ dunif(0, 10)        # Maximum rate at the optimum temperature
  topt ~ dunif(25, 45)       # Optimum temperature (ºC)
  ctmax ~ dunif(35, 50)      # Critical thermal maximum (ºC)
  a ~ dunif(0.1, 10)         # Width parameter
  sigma ~ dunif(0, 1000)     # Standard deviation for the likelihood
  tau <- 1 / (sigma * sigma) # Precision (inverse variance)

  ## Likelihood for observed data
  for (i in 1:N.obs) {
    # Conditional expression for the modified Deutsch function
    trait.mu[i] <- ifelse(
      temp[i] <= topt,
      rmax * exp(-pow((temp[i] - topt) / (2 * a), 2)), # Rising phase
      rmax - rmax * pow((temp[i] - topt) / (topt - ctmax), 2) # Declining phase
    )
    trait[i] ~ dnorm(trait.mu[i], tau) # Observed trait
  }

  ## Predictions for new temperature range
  for (j in 1:N.Temp.xs) {
    r.pred[j] <- ifelse(
      Temp.xs[j] <= topt,
      rmax * exp(-pow((Temp.xs[j] - topt) / (2 * a), 2)), # Rising phase
      rmax - rmax * pow((Temp.xs[j] - topt) / (topt - ctmax), 2) # Declining phase
    )
  }
}
