model {
  # Priors for parameters
  cf.a ~ dunif(0, 10)          # Uniform prior for scaling parameter
  cf.b ~ dunif(0, 1)           # Uniform prior for exponential growth rate
  cf.tmax ~ dunif(20, 50)      # Uniform prior for maximum temperature
  cf.delta_t ~ dunif(0.1, 10)  # Uniform prior for steepness near Tmax
  cf.sigma ~ dunif(0, 5)       # Prior for standard deviation
  cf.tau <- pow(cf.sigma, -2)  # Precision (inverse variance)

  # Likelihood
  for (i in 1:N.obs) {
    trait.mu[i] <- cf.a * exp(cf.b * temp[i]) * 
                   (1 - exp((temp[i] - cf.tmax) / cf.delta_t)) *
                   step(cf.tmax - temp[i])  # Ensure rates decline beyond Tmax
    trait[i] ~ dnorm(trait.mu[i], cf.tau)  # Observed trait
  }

  # Predictions for new temperatures
  for (j in 1:N.Temp.xs) {
    r.pred[j] <- cf.a * exp(cf.b * Temp.xs[j]) * 
                 (1 - exp((Temp.xs[j] - cf.tmax) / cf.delta_t)) *
                 step(cf.tmax - Temp.xs[j])  # Predictions are 0 beyond Tmax
  }
}
